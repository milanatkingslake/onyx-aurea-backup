Remarks

When persisting a scriptSession object in advanceStep, the script engine adds the stateXml property based on the contents of the run-time XML variables section. The scriptSession element of the run time XML should not contain a stateXml node.

Sample

The following XML shows two views of the same run-time XML structure. The "unescaped" view shows the run-time XML in a more user readable format, while the "escaped" view shows entity encoded run-time XML.

"unescaped" sample

<scriptSessionXml>

      <root>

         <script primaryId="100002" initialStepId="100006"/>

         <scriptSession objectType="scriptSession" content="all" action="update">

            <primaryId>100005</primaryId>

            <scriptId>100002</scriptId>

            <ownerId>1</ownerId>

            <ownerType>2</ownerType>

            <ownerSecondaryId>1</ownerSecondaryId>

            <contactId/>

            <contactObjectType/>

            <duration>1</duration>

            <statusId>2</statusId>

            <assignedTo>sa</assignedTo>

            <scriptSessionSteps objectType="scriptSessionStep" collection="1">

               <scriptSessionStep objectType="scriptSessionStep" content="all" action="insert">

                  <primaryId/>

                  <scriptSessionId>100005</scriptSessionId>

                  <scriptStepId>100006</scriptStepId>

                  <sequenceNumber>1</sequenceNumber>

                  <completed>1</completed>

                  <scriptSessionPromptResponses objectType="scriptSessionPromptResponse" collection="1">

                     <scriptSessionPromptResponse objectType="scriptSessionPromptResponse" content="all" action="insert">

                        <primaryId/>

                        <scriptSessionStepId/>

                        <scriptPromptId>100003</scriptPromptId>

                        <responseCode>6</responseCode>

                        <value>von Amsterdam, Livonia</value>

                        <additionalText null="1"/>

                     </scriptSessionPromptResponse>

                     <scriptSessionPromptResponse objectType="scriptSessionPromptResponse" content="all" action="insert">

                        <primaryId/>

                        <scriptSessionStepId/>

                        <scriptPromptId>100004</scriptPromptId>

                        <responseCode>continueYes</responseCode>

                        <value>Continue</value>

                        <additionalText null="1"/>

                     </scriptSessionPromptResponse>

                  </scriptSessionPromptResponses>

               </scriptSessionStep>

            </scriptSessionSteps>

         </scriptSession>

         <variables>

            <variable name="_ownerid" type="scalar">

               <item text="1" code="1"/>

            </variable>

            <variable name="_ownertype" type="scalar">

               <item text="2" code="2"/>

            </variable>

            <variable name="_ownername" type="scalar">

               <item text="Porter Power Washer" code="Porter Power Washer"/>

            </variable>

            <variable name="_userid" type="scalar">

               <item text="System Administrator" code="sa"/>

            </variable>

            <variable name="_scriptsessionid" type="scalar">

               <item text="100005" code="100005"/>

            </variable>

         </variables>

      </root>

</scriptSessionXml>
